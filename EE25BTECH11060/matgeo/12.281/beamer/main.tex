\begin{frame}{Question}
$\Vec{A}$ is a $2 \times 2$ matrix with det $\Vec{A}=2$ .Then det 2$\Vec{A}$  is
\end{frame}
\begin{frame}{Solution}
  As
\begin{align}
\det(kA) = k^n \det(A)
\end{align}
Here, $n = 2$ and $k = 2$. Thus,
\begin{align}
\det(2A) = 2^2 \cdot \det(A) \\  = 4 \cdot 2 \\ = 8
\end{align}  
\end{frame}

}
